28      private JFrame runko;
29      private Syntetisaattori syntikka;
30      private AudioScope nakyma;
31      private UnitOscillator osk1;
32      private UnitOscillator osk2;
33      private UnitOscillator osk3;
34      private double taajuus1, voimakkuus1, taajuus2, voimakkuus2, taajuus3, voimakkuus3;
35      private SoitinGUI sG;

37      /**
38       * Alustaa muokkaus-gui:n.
39       *
40       * @param syntikka Syntetisaattori, joka on käytössä.
41       * @param osk1 Ensimmäinen oskillaattori.
42       * @param osk2 Toinen oskillaattori.
43       * @param osk3 Kolmas oskillaattori.
44       */
45      public GUI(Syntetisaattori syntikka, UnitOscillator osk1, UnitOscillator osk2, UnitOscillator osk3) {
46          this.syntikka = syntikka;
47          nakyma = new AudioScope(syntikka.getMasiina());
48          this.osk1 = osk1;
49          this.osk2 = osk2;
50          this.osk3 = osk3;
51          taajuus1 = osk1.frequency.get();
52          voimakkuus1 = osk1.amplitude.get();
53          taajuus2 = osk2.frequency.get();
54          voimakkuus2 = osk2.amplitude.get();
55          taajuus3 = osk3.frequency.get();
56          voimakkuus3 = osk3.amplitude.get();
57          sG = null;
58      }
59  
60      @Override
61      public void run() {
62          runko = new JFrame("Masiina");
63          runko.setPreferredSize(new Dimension(1200, 600));
64          runko.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
65          luoKomponentit(runko.getContentPane(), nakyma);
66          runko.pack();
67          runko.setVisible(true);
68      }

70      private void luoKomponentit(Container sisalto, AudioScope aS) {
71          GridLayout layout = new GridLayout(8, 1);
72          sisalto.setLayout(layout);
73          aS.addProbe(syntikka.getOsk1().output);
74          aS.addProbe(syntikka.getOsk2().output);
75          aS.addProbe(syntikka.getOsk3().output);
76          aS.setViewMode(AudioScope.ViewMode.SPECTRUM);
77          aS.setTriggerMode(AudioScope.TriggerMode.AUTO);
78          aS.setTriggerLevel(Double.MIN_VALUE);
79          aS.getView().setControlsVisible(false);
80          JButton aloita, lopeta;
81          aloita = new JButton("Aloita");
82          lopeta = new JButton("Jatka");
83          ButtonKuuntelija kuuntelija = new ButtonKuuntelija(syntikka, aloita, lopeta, aS, sG, runko);
84          aloita.addActionListener(kuuntelija);
85          lopeta.addActionListener(kuuntelija);
86          SpinnerNumberModel model1 = new SpinnerNumberModel(440.0, 0.00, 8000.00, 0.01);
87          SpinnerNumberModel model2 = new SpinnerNumberModel(440.0, 0.00, 8000.00, 0.01);
88          SpinnerNumberModel model3 = new SpinnerNumberModel(440.0, 0.00, 8000.00, 0.01);
89          JLabel taajuusOsk1 = new JLabel("1. Oskillaattorin Taajuus: " + taajuus1 + " Hz");
90          JSpinner taajuusSpinner1 = new JSpinner(model1);
91          JSlider taajuusSlider1 = new JSlider(JSlider.HORIZONTAL, 0, 790213, 44000);
92          JLabel voimakkuusOsk1 = new JLabel("1. Oskillaattorin Voimakkuus: " + voimakkuus1);
93          JSlider voimakkuusSlider1 = new JSlider(JSlider.HORIZONTAL, 0, 10, 0);
94          JLabel taajuusOsk2 = new JLabel("2. Oskillaattorin Taajuus: " + taajuus2 + " Hz");
95          JSpinner taajuusSpinner2 = new JSpinner(model2);
96          JSlider taajuusSlider2 = new JSlider(JSlider.HORIZONTAL, 0, 790213, 44000);
97          JLabel voimakkuusOsk2 = new JLabel("2. Oskillaattorin Voimakkuus: " + voimakkuus2);
98          JSlider voimakkuusSlider2 = new JSlider(JSlider.HORIZONTAL, 0, 10, 0);
99          JLabel taajuusOsk3 = new JLabel("3. Oskillaattori Taajuus: " + taajuus3 + " Hz");
100         JSpinner taajuusSpinner3 = new JSpinner(model3);
101         JSlider taajuusSlider3 = new JSlider(JSlider.HORIZONTAL, 0, 790213, 44000);
102         JLabel voimakkuusOsk3 = new JLabel("3. Oskillaattorin Voimakkuus: " + voimakkuus3);
103         JSlider voimakkuusSlider3 = new JSlider(JSlider.HORIZONTAL, 0, 10, 0);
104         TaajuusSliderKuuntelija tSK1 = new TaajuusSliderKuuntelija(1, taajuusOsk1, taajuusSlider1, osk1);
105         taajuusSlider1.addChangeListener(tSK1);
106         TaajuusSliderKuuntelija tSK2 = new TaajuusSliderKuuntelija(2, taajuusOsk2, taajuusSlider2, osk2);
107         taajuusSlider2.addChangeListener(tSK2);
108         TaajuusSliderKuuntelija tSK3 = new TaajuusSliderKuuntelija(3, taajuusOsk3, taajuusSlider3, osk3);
109         taajuusSlider3.addChangeListener(tSK3);
110         VoimakkuusSliderKuuntelija vSK1 = new VoimakkuusSliderKuuntelija(1, voimakkuusOsk1, voimakkuusSlider1, osk1);
111         voimakkuusSlider1.addChangeListener(vSK1);
112         VoimakkuusSliderKuuntelija vSK2 = new VoimakkuusSliderKuuntelija(2, voimakkuusOsk2, voimakkuusSlider2, osk2);
113         voimakkuusSlider2.addChangeListener(vSK2);
114         VoimakkuusSliderKuuntelija vSK3 = new VoimakkuusSliderKuuntelija(3, voimakkuusOsk3, voimakkuusSlider3, osk3);
115         voimakkuusSlider3.addChangeListener(vSK3);
116         sisalto.add(luoValikko1(aloita, lopeta));
117         sisalto.add(luoValikko2(taajuusOsk1, taajuusSlider1, taajuusSpinner1));
118         sisalto.add(luoValikko3(voimakkuusOsk1, voimakkuusSlider1));
119         sisalto.add(luoValikko2(taajuusOsk2, taajuusSlider2, taajuusSpinner2));
120         sisalto.add(luoValikko3(voimakkuusOsk2, voimakkuusSlider2));
121         sisalto.add(luoValikko2(taajuusOsk3, taajuusSlider3, taajuusSpinner3));
122         sisalto.add(luoValikko3(voimakkuusOsk3, voimakkuusSlider3));
123         sisalto.add(aS.getView());
124     }

126     private JPanel luoValikko1(JButton b1, JButton b2) {
127         JPanel paneeli = new JPanel(new GridLayout(2, 1));
128         JPanel paneeli2 = new JPanel(new GridLayout(1, 2));
129         JTextField text = new JTextField("1. Säädä tietokoneen volyymi mahdollisimman pienelle!"
130                 + " 2. Paina aloita"
131                 + " 3. Säädä oskillaattoreiden voimakkuutta/taajuutta");
132         text.disable();
133         paneeli2.add(b1);
134         paneeli2.add(b2);
135         paneeli.add(text);
136         paneeli.add(paneeli2);
137         return paneeli;
138     }

148         paneeli3.add(label);
149         paneeli3.add(label);
150         paneeli2.add(paneeli3);
151         paneeli.add(paneeli2);
152         paneeli.add(jS);
153         return paneeli;
154     }
