3   import com.jsyn.scope.AudioScope;
4   import java.awt.event.ActionEvent;
5   import java.awt.event.ActionListener;
6   import java.util.logging.Level;
7   import java.util.logging.Logger;
8   import javax.swing.JButton;
9   import javax.swing.JFrame;
10  import lesktimo.oskillaattori.aani.Syntetisaattori;
11  import lesktimo.oskillaattori.gui.SoitinGUI;
12  
13  /**
14   * ButtonKuuntelija kuuntelee GUI:n nappeja.
15   */
16  public class ButtonKuuntelija implements ActionListener {
17  
18      private Syntetisaattori syntikka;
19      private AudioScope aS;
20      private JButton aloita;
21      private JButton jatka;
22      private SoitinGUI sG;
23      private JFrame vanhaRunko;
24  
25      /**
26       * Alustaa ButtonKuuntelijan.
27       *
28       * @param syntikka Käytössä oleva syntetisaattori
29       * @param aloita JButton, joka aloittaa ja keskeyttää soiton.
30       * @param jatka JButton, joka siirtyy koskettimiston sisältävään näkymään.
31       * @param aS Waveform-näkymä
32       * @param sG SoitinGUI, jonka nappi luo.
33       * @param runko Vanha JFrame-runko, joka piilotetaan uusi luodessa.
34       */
35      public ButtonKuuntelija(Syntetisaattori syntikka, JButton aloita, JButton jatka, AudioScope aS, SoitinGUI sG, JFrame runko) {
36          this.syntikka = syntikka;
37          this.aloita = aloita;
38          this.jatka = jatka;
39          this.aS = aS;
40          this.sG = sG;
41          this.vanhaRunko = runko;
42      }
43  
44      /**
45       * Mikäli käyttäjä painaa aloita, alkavat oskillaatorit soida, ja nappi
46       * vaihtuu pysäytä -napiksi, joka puolestaan pysäyttää oskillaarit.
47       * Jatka-nappi siirtyy SoitinGUI:hin.
48       *
49       */
50      @Override
51      public void actionPerformed(ActionEvent e) {
52          if (e.getSource().equals(aloita)) {
53              if (syntikka.isOn() == false) {
54                  try {
55                      syntikka.aloita();
56                      aS.start();
57                      aloita.setText("Pysäytä");
58                  } catch (InterruptedException ex) {
59                      Logger.getLogger(ButtonKuuntelija.class.getName()).log(Level.SEVERE, null, ex);
60                  }
61              } else {
62                  syntikka.lopeta();
63                  aS.stop();
64                  aloita.setText("Aloita");
65              }
66          } else if (e.getSource().equals(jatka)) {
67  //            if (syntikka.isOn() == true) {
68  //                syntikka.lopeta();
69  
70  //            }
71              try {
72                  syntikka.yhdistaAanet();
73                  sG = new SoitinGUI(syntikka, aS);
74                  sG.run();
75                  vanhaRunko.hide();
76              } catch (InterruptedException ex) {
77                  Logger.getLogger(ButtonKuuntelija.class.getName()).log(Level.SEVERE, null, ex);
78              }
79          }
80      }
81  }
